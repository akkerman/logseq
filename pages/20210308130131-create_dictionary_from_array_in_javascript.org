#+title: Create dictionary from array in JavaScript

function arrayToDict ({ keyProp, valueProp = null, array }) {
  const reducer = valueProp
    ? (result, item) => ({ ...result, [item[keyProp]]: item[valueProp] })
    : (result, item) => ({ ...result, [item[keyProp]]: item })

  return array.reduce(reducer, {}) // empty object as start value
}
